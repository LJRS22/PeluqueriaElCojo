IF EXISTS (SELECT name FROM sys.databases WHERE name = 'PeluqueriaElCojo')

    ALTER DATABASE PeluqueriaElCojo SET SINGLE_USER WITH ROLLBACK IMMEDIATE;
    DROP DATABASE PeluqueriaElCojo;
END
GO

CREATE DATABASE PeluqueriaElCojo;
GO

USE PeluqueriaElCojo;
GO

-- ═══════════════════════════════════════════════
-- TABLAS
-- ═══════════════════════════════════════════════

CREATE TABLE Clientes (
    Id INT IDENTITY(1,1) PRIMARY KEY,
    Nombre NVARCHAR(50) NOT NULL,
    Telefono VARCHAR(10) NOT NULL,
    Tipo INT NOT NULL DEFAULT 0,
    Visitas INT NOT NULL DEFAULT 0,
    FechaRegistro DATETIME NOT NULL DEFAULT GETDATE(),
    Activo BIT NOT NULL DEFAULT 1
);
GO

CREATE TABLE Empleados (
    Id INT IDENTITY(1,1) PRIMARY KEY,
    Nombre NVARCHAR(50) NOT NULL,
    Apodo NVARCHAR(20) NULL,
    Cedula VARCHAR(13) NOT NULL UNIQUE,
    Telefono VARCHAR(10) NULL,
    Rol INT NOT NULL DEFAULT 0,
    SalarioBase DECIMAL(10,2) NOT NULL DEFAULT 20000,
    PorcentajeComision DECIMAL(5,2) NOT NULL DEFAULT 10,
    FechaIngreso DATETIME NOT NULL DEFAULT GETDATE(),
    VentasMes DECIMAL(12,2) NOT NULL DEFAULT 0,
    Activo BIT NOT NULL DEFAULT 1
);
GO

CREATE TABLE Productos (
    Id INT IDENTITY(1,1) PRIMARY KEY,
    Codigo VARCHAR(20) NOT NULL UNIQUE,
    Nombre NVARCHAR(50) NOT NULL,
    Categoria INT NOT NULL DEFAULT 6,
    Precio DECIMAL(10,2) NOT NULL,
    Costo DECIMAL(10,2) NOT NULL,
    Stock INT NOT NULL DEFAULT 0,
    StockMinimo INT NOT NULL DEFAULT 5,
    Activo BIT NOT NULL DEFAULT 1
);
GO

CREATE TABLE Servicios (
    Id INT IDENTITY(1,1) PRIMARY KEY,
    Nombre NVARCHAR(50) NOT NULL,
    PrecioBase DECIMAL(10,2) NOT NULL,
    DuracionMinutos INT NOT NULL DEFAULT 20,
    Activo BIT NOT NULL DEFAULT 1
);
GO

CREATE TABLE Facturas (
    Id INT IDENTITY(1,1) PRIMARY KEY,
    Fecha DATETIME NOT NULL DEFAULT GETDATE(),
    ClienteId INT NOT NULL FOREIGN KEY REFERENCES Clientes(Id),
    EmpleadoId INT NOT NULL FOREIGN KEY REFERENCES Empleados(Id),
    Subtotal DECIMAL(12,2) NOT NULL,
    Descuento DECIMAL(12,2) NOT NULL DEFAULT 0,
    ITBIS DECIMAL(12,2) NOT NULL,
    Total DECIMAL(12,2) NOT NULL
);
GO

CREATE TABLE Usuarios (
    Id INT IDENTITY(1,1) PRIMARY KEY,
    Usuario NVARCHAR(50) NOT NULL UNIQUE,
    Contrasena NVARCHAR(50) NOT NULL,
    Rol INT NOT NULL DEFAULT 1,
    Activo BIT NOT NULL DEFAULT 1
);
GO
